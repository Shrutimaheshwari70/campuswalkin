      function starsHtml(r) {
        const f = Math.round(r * 2) / 2;
        let s = "";
        for (let i = 1; i <= 5; i++) {
          s += i <= f ? "★" : i - 0.5 === f ? "½" : "☆";
        }
        return s;
      }

      function pct(val) {
        return Math.round((val / 5) * 100);
      }
