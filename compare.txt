        { id:'lknu',  name:'Univ. of Lucknow',     icon:'🏛️', type:'State',      rank:'#42', fee:'₹10K/yr' },
        { id:'bbau',  name:'Babasaheb Ambedkar U.',icon:'📖', type:'Central',    rank:'#36', fee:'₹14K/yr' },
      ],
    }
  };

  const STEPS = [
    { title: 'Choose a City', desc: 'Select the city where you want to study' },
    { title: 'Choose a University', desc: '' },
    { title: 'Program Details', desc: '' },
  ];

  let state = { step: 0, city: null, university: null };
  let addedPrograms = [];

  function openStepper() {
    if (addedPrograms.length >= 4) return alert('Maximum 4 programs can be compared.');
    state = { step: 0, city: null, university: null };
    document.getElementById('stepperPanel').style.display = 'block';
    document.getElementById('addCard').style.display = 'none';
    render();
  }

  function closeStepper() {
    document.getElementById('stepperPanel').style.display = 'none';
    document.getElementById('addCard').style.display = 'flex';
  }

  function render() {
    renderProgress();
    if (state.step === 0) renderCityStep();
    else if (state.step === 1) renderUniStep();
    else renderDetailStep();
  }

  function renderProgress() {
    const track = document.getElementById('progressTrack');
    track.innerHTML = STEPS.map((s, i) => {
      const cls = i < state.step ? 'done' : i === state.step ? 'active' : '';
      const icon = i < state.step
        ? '<i class="bi bi-check-lg" style="font-size:13px;"></i>'
        : (i + 1);
      const lineCls = i < state.step ? 'done' : '';
      return `
        <div class="progress-step ${cls}">
          <div class="progress-dot ${cls}">${icon}</div>
          <div class="progress-label">${s.title.split(' ')[1] || s.title}</div>
        </div>
        ${i < STEPS.length - 1 ? `<div class="progress-line ${lineCls}"></div>` : ''}
      `;
    }).join('');

    const stepTitleEl = document.getElementById('stepTitle');
    const stepDescEl  = document.getElementById('stepDesc');
    const city = state.city ? DATA.cities.find(c => c.id === state.city) : null;
    if (state.step === 1 && city) {
      stepTitleEl.textContent = `Universities in ${city.name}`;
      stepDescEl.textContent  = `${city.state} — ${city.count} universities available`;
    } else {
      stepTitleEl.textContent = STEPS[state.step].title;
      stepDescEl.textContent  = STEPS[state.step].desc;
    }
  }

  /* ──── Step 0: City ──── */
  function renderCityStep() {
    document.getElementById('stepBody').innerHTML = `
      <div class="search-wrap">
        <i class="bi bi-search"></i>
        <input class="search-inp" id="citySearch" placeholder="Search cities or states…" oninput="filterCities(this.value)" />
      </div>
      <div class="opts-grid" id="cityGrid"></div>
    `;
    renderCityGrid('');
    renderFooter();
  }

  function filterCities(q) { renderCityGrid(q); }

  function renderCityGrid(q) {
    q = q.toLowerCase();
    const list = DATA.cities.filter(c =>
      c.name.toLowerCase().includes(q) || c.state.toLowerCase().includes(q)
    );
    document.getElementById('cityGrid').innerHTML = list.map(c => `
      <div class="opt-tile ${state.city === c.id ? 'selected' : ''}" onclick="selectCity('${c.id}')">
        <div class="check-badge"><i class="bi bi-check-lg"></i></div>
        <span class="tile-icon">${c.icon}</span>
        <div class="tile-name">${c.name}</div>
        <div class="tile-meta">${c.count} universities</div>
      </div>
    `).join('') || '<div style="color:var(--ink-4);font-size:13px;padding:8px 0;">No results found</div>';
    updateNextBtn();
  }

  function selectCity(id) {
    state.city = id;
    state.university = null;
    renderCityGrid((document.getElementById('citySearch') || {value:''}).value);
  }

  /* ──── Step 1: University ──── */
  function renderUniStep() {
    document.getElementById('stepBody').innerHTML = `
      <div class="search-wrap">
        <i class="bi bi-search"></i>
        <input class="search-inp" id="uniSearch" placeholder="Search universities…" oninput="filterUnis(this.value)" />
      </div>
      <div class="opts-grid" id="uniGrid"></div>
    `;
    renderUniGrid('');
    renderFooter();
  }

  function filterUnis(q) { renderUniGrid(q); }

  function renderUniGrid(q) {
    q = q.toLowerCase();
    const list = (DATA.universities[state.city] || []).filter(u =>
      u.name.toLowerCase().includes(q) || u.type.toLowerCase().includes(q)
    );
    document.getElementById('uniGrid').innerHTML = list.map(u => `
      <div class="opt-tile ${state.university === u.id ? 'selected' : ''}" onclick="selectUni('${u.id}')">
        <div class="check-badge"><i class="bi bi-check-lg"></i></div>
        <span class="tile-icon">${u.icon}</span>
        <div class="tile-name">${u.name}</div>
        <div class="tile-meta">${u.type} · ${u.rank}</div>
      </div>
    `).join('') || '<div style="color:var(--ink-4);font-size:13px;padding:8px 0;">No results found</div>';
    updateNextBtn();
  }

  function selectUni(id) {
    state.university = id;
    renderUniGrid((document.getElementById('uniSearch') || {value:''}).value);
  }

  /* ──── Step 2: Details ──── */
  function renderDetailStep() {
    const city = DATA.cities.find(c => c.id === state.city);
    const uni  = (DATA.universities[state.city] || []).find(u => u.id === state.university);
    const programs = ['MBA', 'B.Tech', 'M.Tech', 'BBA', 'PhD', 'MCA'];
    const modes    = ['Full-time', 'Part-time', 'Online', 'Executive'];

    document.getElementById('stepBody').innerHTML = `
      <div class="result-hero">
        <div class="result-avatar">${uni.icon}</div>
        <div>
          <div class="result-name">${uni.name}</div>
          <div class="result-location">
            <i class="bi bi-geo-alt" style="font-size:12px;"></i>
            ${city.icon} ${city.name}, ${city.state}
          </div>
        </div>
      </div>

      <div class="stats-row">
        <div class="stat-box">
          <div class="stat-label">National Rank</div>
          <div class="stat-value">${uni.rank}</div>
        </div>
        <div class="stat-box">
          <div class="stat-label">Institute Type</div>
          <div class="stat-value" style="font-size:14px;">${uni.type}</div>
        </div>
        <div class="stat-box">
          <div class="stat-label">Avg. Fee</div>
          <div class="stat-value" style="font-size:14px;">${uni.fee}</div>
        </div>
      </div>

      <div class="tags-section">
        <div class="tags-label">Programs Offered</div>
        <div class="tags-row">${programs.map(p => `<span class="tag blue">${p}</span>`).join('')}</div>
      </div>

      <div class="tags-section">
        <div class="tags-label">Study Mode</div>
        <div class="tags-row">${modes.map(m => `<span class="tag gold">${m}</span>`).join('')}</div>
      </div>

      <button class="btn-add" onclick="addToCompare()">
        <i class="bi bi-plus-circle-fill"></i> Add to Comparison
      </button>
    `;
    renderFooter();
  }

  /* ──── Footer ──── */
  function renderFooter() {
    const city = state.city ? DATA.cities.find(c => c.id === state.city) : null;
    const uni  = state.university ? (DATA.universities[state.city] || []).find(u => u.id === state.university) : null;

    const crumbs = [];
    if (city) crumbs.push(`${city.icon} ${city.name}`);
    if (uni)  crumbs.push(uni.name);

    document.getElementById('stepFooter').innerHTML = `
      <div class="breadcrumb-trail">
        ${crumbs.map((c, i) => `
          ${i > 0 ? '<i class="bi bi-chevron-right crumb-sep" style="font-size:10px;"></i>' : ''}
          <span>${c}</span>
        `).join('')}
      </div>
      <div style="display:flex;gap:8px;">
        ${state.step > 0 ? `<button class="btn-back" onclick="goBack()"><i class="bi bi-arrow-left" style="font-size:11px;"></i> Back</button>` : `<button class="btn-back" onclick="closeStepper()">Cancel</button>`}
        ${state.step < 2 ? `<button class="btn-next" id="nextBtn" onclick="goNext()" ${!canProceed() ? 'disabled' : ''}><span>Next</span><i class="bi bi-arrow-right" style="font-size:12px;"></i></button>` : ''}
      </div>
    `;
  }

  function updateNextBtn() {
    const btn = document.getElementById('nextBtn');
    if (btn) btn.disabled = !canProceed();
    renderFooter();
  }

  function canProceed() {
    if (state.step === 0) return !!state.city;
    if (state.step === 1) return !!state.university;
    return false;
  }

  function goNext() {
    if (!canProceed()) return;
    state.step++;
    render();
  }

  function goBack() {
    if (state.step > 0) { state.step--; render(); }
  }

  /* ──── Add to Compare ──── */
  function addToCompare() {
    const city = DATA.cities.find(c => c.id === state.city);
    const uni  = (DATA.universities[state.city] || []).find(u => u.id === state.university);
    const already = addedPrograms.find(p => p.uniId === state.university);
    if (already) { alert('This university is already in your comparison.'); return; }

    addedPrograms.push({ cityId: state.city, uniId: state.university, city, uni });
    renderAddedCards();
    closeStepper();
  }

  function removeProgram(uniId) {
    addedPrograms = addedPrograms.filter(p => p.uniId !== uniId);
    renderAddedCards();
    if (addedPrograms.length < 4) {
      document.getElementById('addCard').style.display = 'flex';
    }
  }

  function renderAddedCards() {
    const existing = document.querySelectorAll('.added-card');
    existing.forEach(el => el.remove());

    const row = document.getElementById('compareRow');
    const addCard = document.getElementById('addCard');

    addedPrograms.forEach(p => {
      const card = document.createElement('div');
      card.className = 'added-card';
      card.innerHTML = `
        <button class="remove-btn" onclick="removeProgram('${p.uniId}')"><i class="bi bi-x"></i></button>
        <div class="card-icon">${p.uni.icon}</div>
        <div class="card-uni">${p.uni.name}</div>
        <div class="card-city">${p.city.icon} ${p.city.name}</div>
        <div class="card-rank"><i class="bi bi-trophy-fill" style="font-size:10px;"></i> ${p.uni.rank}</div>
        <div style="margin-top:6px;font-size:11px;color:var(--ink-3);">${p.uni.type} · ${p.uni.fee}</div>
      `;
      row.insertBefore(card, addCard);
    });

    if (addedPrograms.length >= 4) {
      addCard.style.display = 'none';
    }
  }
</script>
